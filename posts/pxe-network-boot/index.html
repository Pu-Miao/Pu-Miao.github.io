PXE技术概述#

PXE（Preboot Execution Environment，预启动执行环境）是一种网络启动技术，允许计算机通过网络引导和安装操作系统，无需本地存储介质。这对于批量部署服务器、无盘工作站和系统恢复非常有用。

    Client->>DHCP: 1. DHCP请求
    DHCP->>Client: 2. IP地址 + TFTP服务器信息
    Client->>TFTP: 3. 请求引导文件
    TFTP->>Client: 4. 发送引导文件
    Client->>HTTP: 5. 请求安装文件
    HTTP->>Client: 6. 传输系统文件

服务器要求#

  • 操作系统: Ubuntu 20.04+ / CentOS 8+
  • 内存: 至少4GB RAM
  • 存储: 至少100GB可用空间
  • 网络: 千兆网卡，独立网段管理

# /etc/dhcp/dhcpd.conf
option domain-name "pxe.local";
option domain-name-servers 8.8.8.8, 114.114.114.114;

default-lease-time 600;
max-lease-time 7200;

# PXE配置选项
option space PXE;
option PXE.mtftp-ip    code 1 = ip-address;
option PXE.mtftp-cport code 2 = unsigned integer 16;
option PXE.mtftp-sport code 3 = unsigned integer 16;
option PXE.mtftp-tmout code 4 = unsigned integer 8;
option PXE.mtftp-delay code 5 = unsigned integer 8;
vendor-option-space PXE;

# DHCP作用域配置
subnet 192.168.100.0 netmask 255.255.255.0 {
    range 192.168.100.100 192.168.100.200;
    option routers 192.168.100.1;
    option broadcast-address 192.168.100.255;
    
    # PXE启动配置
    next-server 192.168.100.10;  # TFTP服务器地址
    filename "pxelinux.0";       # 引导文件名
    
    # UEFI启动支持
    if exists user-class and option user-class = "iPXE" {
        filename "http://192.168.100.10/boot.ipxe";
    } elsif option arch = encode-int(16, 16) {
        filename "bootx64.efi";
    } elsif option arch = encode-int(0, 16) {
        filename "pxelinux.0";
    }
}

# 静态IP分配示例
host server01 {
    hardware ethernet 00:11:22:33:44:55;
    fixed-address 192.168.100.11;
}

#!/bin/bash
# PXE服务监控脚本

while true; do
    clear
    echo "=== PXE Server Status Monitor ==="
    echo "Time: $(date)"
    echo
    
    # 检查服务状态
    echo "=== Service Status ==="
    systemctl is-active dhcp-server && echo "DHCP: Running" || echo "DHCP: Stopped"
    systemctl is-active tftpd-hpa && echo "TFTP: Running" || echo "TFTP: Stopped"
    systemctl is-active nginx && echo "HTTP: Running" || echo "HTTP: Stopped"
    echo
    
    # 显示活动连接
    echo "=== Active TFTP Connections ==="
    netstat -un | grep :69
    echo
    
    # 显示最近的DHCP分配
    echo "=== Recent DHCP Leases ==="
    tail -5 /var/lib/dhcp/dhcpd.leases | grep -E "(lease|client-hostname|hardware ethernet)"
    echo
    
    sleep 5
done

#!/bin/bash
# PXE批量部署脚本

# 配置变量
PXE_ROOT="/var/lib/tftpboot"
HTTP_ROOT="/var/www/html"
TFTP_SERVER="192.168.100.10"

# 函数：创建主机专用配置
create_host_config() {
    local hostname=$1
    local mac_addr=$2
    local install_type=$3
    
    # 转换MAC地址格式
    formatted_mac=$(echo $mac_addr | sed 's/:/-/g' | tr '[:upper:]' '[:lower:]')
    config_file="$PXE_ROOT/pxelinux.cfg/01-$formatted_mac"
    
    # 创建专用配置
    cat > $config_file << EOF
DEFAULT $install_type
TIMEOUT 30

LABEL $install_type
    KERNEL images/ubuntu-vmlinuz
    APPEND initrd=images/ubuntu-initrd ip=dhcp url=http://$TFTP_SERVER/images/ubuntu autoinstall ds=nocloud-net;s=http://$TFTP_SERVER/kickstart/$hostname/
EOF
    
    echo "Created config for $hostname ($mac_addr)"
}

# 函数：创建主机专用kickstart
create_host_kickstart() {
    local hostname=$1
    local ip_addr=$2
    
    mkdir -p "$HTTP_ROOT/kickstart/$hostname"
    
    # 复制并修改user-data
    sed "s/ubuntu-server/$hostname/g" "$HTTP_ROOT/kickstart/ubuntu/user-data" > "$HTTP_ROOT/kickstart/$hostname/user-data"
    cp "$HTTP_ROOT/kickstart/ubuntu/meta-data" "$HTTP_ROOT/kickstart/$hostname/meta-data"
    
    echo "Created kickstart for $hostname"
}

# 主机列表文件格式：hostname,mac_address,ip_address,install_type
while IFS=',' read -r hostname mac_addr ip_addr install_type; do
    [[ $hostname =~ ^#.*$ ]] && continue  # 跳过注释行
    
    echo "Processing $hostname..."
    create_host_config "$hostname" "$mac_addr" "$install_type"
    create_host_kickstart "$hostname" "$ip_addr"
    
done < /etc/pxe-hosts.csv

echo "Batch deployment configuration completed!"

    # 启用gzip压缩
    gzip on;
    gzip_types text/plain application/xml text/css application/javascript;
    
    # 设置缓存头
    location ~* \.(iso|img|vmlinuz|initrd)$ {
        expires 1d;
        add_header Cache-Control "public, immutable";
    }
    
    # 限制上传大小
    client_max_body_size 100M;
