计算环境部署文档

Ubuntu#

基础环境准备#

系统优化#

锁内核：

apt-mark hold linux-generic linux-image-generic linux-headers-generic

关闭自动更新：

sudo sed -i.bak 's/1/0/' /etc/apt/apt.conf.d/10periodic &> /dev/null

sudo sed -i.bak 's/1/0/' /etc/apt/apt.conf.d/20auto-upgrades &> /dev/null

依赖安装#

apt install gcc make -y

Nvidia fabricmanager 下载（根据实际操作系统修改/SXM需要，PCIE不需要）#

wget https://developer.download.nvidia.cn/compute/cuda/repos/ubuntu2004/x86_64/nvidia-fabricmanager-535_535.154.05-1_amd64.deb

驱动安装#

Nvidia Drvie(根据实际版本修改)#

./NVIDIA-Linux-x86_64-535.154.05.run \

--ui=none \ # 禁用图形界面（纯命令行模式）

--no-questions \ # 跳过所有交互式提问（自动确认）

--accept-license \ # 自动接受许可协议

--disable-nouveau \ # 禁用开源驱动 Nouveau

--install-libglvnd # 安装 libglvnd 兼容层

Cuda(根据实际版本修改)#

./cuda_12.2.0_535.154.05_linux.ru \

--silent \ # 静默模式

--toolkit \ # 仅安装CUDA Toolkit

--override \ # 覆盖某些默认设置

--no-drm \ # 安装DRM库，这样可以避免安装GPU驱动程序

--no-opengl-libs # 不安装OpenGL库

Nvidia fabricmanager#

dpkg -i nvidia-fabricmanager-535_535.154.05-1_amd64.deb

服务设置#

Nvidia fabricmanager(NVLINK)#

systemctl daemon-reload
systemctl start nvidia-fabricmanager
systemctl enable nvidia-fabricmanager

Nvidia-persistenced#

cat << EOF > /lib/systemd/system/nvidia-persistenced.service
[Unit]
Description=NVIDIA Persistence Daemon
After=syslog.target

[Service]
Type=forking
PIDFile=/var/run/nvidia-persistenced/nvidia-persistenced.pid
Restart=always
ExecStart=/usr/bin/nvidia-persistenced --verbose
ExecStopPost=/bin/rm -rf /var/run/nvidia-persistenced/*
TimeoutSec=300

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable nvidia-persistenced
systemctl start nvidia-persistenced

Cuda 环境设置(根据实际情况)#

cat << EOF >> /etc/profile

export PATH=/usr/local/cuda/bin:$PATH

export CUDA_HOME=/usr/local/cuda/

export LD_LIBRARY_PATH=/usr/local/cuda/lib64:$LD_LIBRARY_PATH

export LD_LIBRARY_PATH=/usr/local/cuda/targets/x86_64-linux/lib:$LD_LIBRARY_PATH

export LD_LIBRARY_PATH=/usr/local/cuda/targets/x86_64-linux/lib/stubs:$LD_LIBRARY_PATH

EOF

环境加载
source /etc/profile
